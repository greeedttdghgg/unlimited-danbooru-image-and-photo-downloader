import requests
import os

def download_image(image_url, save_dir):
    try:
        response = requests.get(image_url)
        response.raise_for_status()  # Raise an error for bad responses
        file_name = os.path.join(save_dir, image_url.split('/')[-1])
        
        with open(file_name, 'wb') as file:
            file.write(response.content)
        print(f"Downloaded: {file_name}")
    except Exception as e:

### PHP
In this PHP example, cURL is used to download images from a specified URL. The image data is saved locally to ensure it's easily retrievable.

php
<?php

function downloadImage($imageUrl, $saveDir) {
    $ch = curl_init($imageUrl);
    $fileName = $saveDir . '/' . basename($imageUrl);
    
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    $data = curl_exec($ch);
    
    if (curl_getinfo($ch, CURLINFO_HTTP_CODE) == 200) {
        file_put_contents($fileName, $data);
        echo "Downloaded: $fileName\n";
    } else {
        echo "Error downloading $imageUrl: " . curl_error($ch) . "\n";
    }
    
    curl_close($ch);
}

$saveDirectory = "./downloaded_images";
mkdir($saveDirectory, 0777, true);

// Replace with actual image URLs
$imageUrls = ["https://hdstockimages.com/someimage1.jpg", "https://hdstockimages.com/someimage2.jpg"];
foreach ($imageUrls as $url) {
    downloadImage($url, $saveDirectory);
}
?>

const fs = require('fs'); // Uncomment if using Node.js
// const fetch = require('node-fetch'); // Uncomment for Node.js

async function downloadImage(imageUrl, saveDir) {
    const fileName = `${saveDir}/${imageUrl.split('/').pop()}`;
    try {
        const response = await fetch(imageUrl);
        if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
        
        const buffer = await response.buffer(); // Or response.blob() in a browser
        
        fs.writeFileSync(fileName, buffer); // Use `fs` to write in Node.js
        console.log(`Downloaded: ${fileName}`);
    } catch (error) {
        console.log(`Error downloading ${imageUrl}: ${error.message}`);
    }
}
